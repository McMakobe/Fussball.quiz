#!/usr/bin/env python3 """ Baut players.json für das Karriereweg-Quiz aus Wikidata (frei lizenzierte Daten). Ausführen: python3 build_players_wikidata.py Benötigt: Python 3 mit requests (pip install requests) Dauer: einige Minuten (viele kleine Abfragen, absichtlich gedrosselt) Robustheit: Die Kandidaten werden in Bekanntheits-Bändern geladen (statt in einer Riesenabfrage), jede Abfrage wird bei Fehlern automatisch wiederholt, und kaputte Antworten werden toleriert bzw. übersprungen. """ import json import time import sys try: import requests except ImportError: sys.exit("requests fehlt. Bitte zuerst installieren: pip install requests") # ---------------- Einstellungen ---------------- MIN_SITELINKS = 12 # Mindest-Anzahl Wikipedia-Sprachversionen (Bekanntheitsfilter) MIN_STATIONS = 3 # mind. so viele Vereinsstationen mit Einsatzzahl MIN_TOTAL_APPS = 80 # mind. so viele Einsätze insgesamt BATCH = 40 # Spieler pro Detailabfrage PAUSE = 1.0 # Sekunden Pause zwischen Abfragen (fair gegenüber Wikidata) OUTPUT = "players.json" ENDPOINT = "https://query.wikidata.org/sparql" HEADERS = {"User-Agent": "KarrierewegQuiz/1.0 (privates Hobby-Projekt)"} def sparql(query, tries=3): """Abfrage mit automatischen Wiederholungsversuchen bei Netz-/Parse-Fehlern.""" last = None for t in range(tries): try: r = requests.get(ENDPOINT, params={"query": query, "format": "json"}, headers=HEADERS, timeout=120) r.raise_for_status() # strict=False toleriert unmaskierte Steuerzeichen in Namen return json.loads(r.text, strict=False)["results"]["bindings"] except (json.JSONDecodeError, requests.RequestException, KeyError) as e: last = e time.sleep(5 * (t + 1)) raise last def clean(s): """Entfernt Steuerzeichen und Mehrfach-Leerzeichen aus Namen.""" return " ".join(str(s).split()) # ---------------- Schritt 1: Kandidaten in Bekanntheits-Bändern ---------------- # Kleine Antworten pro Band -> kein Abschneiden durch den Server. BANDS = [200, 120, 80, 60, 50, 40, 34, 30, 26, 22, 19, 16, 14, 12, 10, 8] edges = [e for e in BANDS if e > MIN_SITELINKS] edges.append(MIN_SITELINKS) print(f"Hole Fussballer mit >= {MIN_SITELINKS} Sprachversionen (in Bändern) ...") candidates = [] hi = 1000000 for lo in edges: try: rows = sparql(f""" SELECT ?p ?links WHERE {{ ?p wdt:P106 wd:Q937857 ; wikibase:sitelinks ?links . FILTER(?links >= {lo} && ?links < {hi}) }} """) except Exception as e: print(f" Band {lo}-{hi} übersprungen ({e})") hi = lo continue for r in rows: qid = r["p"]["value"].rsplit("/", 1)[-1] candidates.append({"qid": qid, "links": int(r["links"]["value"])}) print(f" Band >= {lo}: total {len(candidates)} Kandidaten") hi = lo time.sleep(PAUSE) # Duplikate ueber die eindeutige Wikidata-ID entfernen seen_qids = set() candidates = [c for c in candidates if not (c["qid"] in seen_qids or seen_qids.add(c["qid"]))] print(f"{len(candidates)} Kandidaten gefunden.") if not candidates: sys.exit("Keine Kandidaten - MIN_SITELINKS senken.") candidates.sort(key=lambda c: -c["links"]) # Quartilsgrenzen fuer die Schwierigkeit links_sorted = sorted(c["links"] for c in candidates) def q(p): return links_sorted[min(len(links_sorted) - 1, int(p * len(links_sorted)))] q25, q50, q75 = q(0.25), q(0.50), q(0.75) def difficulty(links): if links >= q75: return 1 if links >= q50: return 2 if links >= q25: return 3 return 4 # ---------------- Schritt 2: Namen + Karrieren in kleinen Batches ---------------- result = [] seen_players = set() for i in range(0, len(candidates), BATCH): batch = candidates[i:i + BATCH] values = " ".join("wd:" + c["qid"] for c in batch) try: rows = sparql(f""" SELECT ?p ?pLabel ?teamLabel ?apps ?start ?logo WHERE {{ VALUES ?p {{ {values} }} ?p p:P54 ?stmt . ?stmt ps:P54 ?team ; pq:P1350 ?apps . # Anzahl Einsätze (Pflicht) OPTIONAL {{ ?stmt pq:P580 ?start }} # Startdatum fuer die Reihenfolge OPTIONAL {{ ?team wdt:P154 ?logo }} # frei lizenziertes Vereinslogo (falls vorhanden) FILTER NOT EXISTS {{ ?team wdt:P31/wdt:P279* wd:Q6979593 }} # keine Nationalteams SERVICE wikibase:label {{ bd:serviceParam wikibase:language "de,en". }} }} """) except Exception as e: print(f" Batch {i//BATCH + 1} übersprungen ({e})") time.sleep(PAUSE * 3) continue names, per_player = {}, {} for r in rows: qid = r["p"]["value"].rsplit("/", 1)[-1] names[qid] = clean(r.get("pLabel", {}).get("value", "")) team = clean(r["teamLabel"]["value"]) try: apps = int(float(r["apps"]["value"])) except (ValueError, KeyError): continue start = r.get("start", {}).get("value", "9999") logo = r.get("logo", {}).get("value", "") if logo: logo = logo.replace("http://", "https://") + "?width=64" per_player.setdefault(qid, []).append((start, team, apps, logo)) for c in batch: stations = sorted(per_player.get(c["qid"], [])) path = [([team, apps, 0, logo] if logo else [team, apps]) for _, team, apps, logo in stations if apps > 0] total = sum(s[1] for s in path) name = names.get(c["qid"], "") if len(path) < MIN_STATIONS or total < MIN_TOTAL_APPS or not name: continue if name.startswith("Q") and name[1:].isdigit(): continue # Eintrag ohne echten Namen key = (name.lower(), path[0][0], total) if key in seen_players: continue # gleicher Name, gleicher Startverein, gleiche Einsatzsumme -> Duplikat seen_players.add(key) result.append({"d": difficulty(c["links"]), "name": name, "alt": [], "path": path}) done = min(i + BATCH, len(candidates)) print(f" {done}/{len(candidates)} geprueft, {len(result)} Spieler im Pool") time.sleep(PAUSE) if not result: sys.exit("Keine Spieler uebrig - Filter lockern (MIN_SITELINKS/MIN_STATIONS senken).") with open(OUTPUT, "w", encoding="utf-8") as f: json.dump(result, f, ensure_ascii=False, separators=(",", ":")) from collections import Counter dist = Counter(p["d"] for p in result) print(f"\nFertig: {len(result)} Spieler -> {OUTPUT}") print(f"Schwierigkeit: {dist[1]} leicht, {dist[2]} mittel, {dist[3]} schwer, {dist[4]} sehr schwer") print("players.json jetzt neben die index.html ins Repository hochladen.")
